#version 330 core
in vec3 color;
out vec4 FragColor;
float rand(vec2 co) {
    return fract(sin(dot(co.xy, vec2(12.9898,78.233))) * 43758.5453);
}
void main()
{
    vec2 uv = gl_FragCoord.xy / vec2(800.0, 600.0); // Assuming screen size is 800x600
    float randomValue = rand(uv * 10.0); // Adjust multiplier to change randomness frequency
    vec3 modifiedColor = color + vec3(randomValue/256);
    FragColor = vec4(modifiedColor, 1.0);
}
